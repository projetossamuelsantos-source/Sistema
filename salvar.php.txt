<?php header('Content-Type: application/json'); $arquivo = 'historico.json'; $dados = json_decode( file_get_contents("php://input"), true ); if (!$dados) { echo json_encode([ "status" => "erro", "mensagem" => "Dados inválidos" ]); exit; } if (!file_exists($arquivo)) { file_put_contents($arquivo, '[]'); } $historico = json_decode( file_get_contents($arquivo), true ); if (!$historico) { $historico = []; } array_unshift($historico, $dados); file_put_contents( $arquivo, json_encode( $historico, JSON_PRETTY_PRINT | JSON_UNESCAPED_UNICODE ) ); echo json_encode([ "status" => "ok" ]); ?>
